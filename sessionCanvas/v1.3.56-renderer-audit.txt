function renderMonsterCardsHtml(node) {
  if (node.type !== 'encounter') return '';
  const list = node.fields && Array.isArray(node.fields.monsters) ? node.fields.monsters : [];
  if (!list.length) return '';
  const groups = groupImportedMonsters(list);
  return `
    <div class="node-monsters">
      ${groups.map(g => renderMonsterGroupHtml(g, node)).join('')}
    </div>`;
}
