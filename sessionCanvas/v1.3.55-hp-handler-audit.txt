// Always-visible HP bar — sits between the monster's header (letter/
// flavor/name/peek-toggle) and its action buttons. Pulled out of the
// peek body in Stage 29 so HP is one click away during play.
//
// Input accepts:
//   • plain number      → set absolute HP (e.g. "12" → hp_current = 12)
//   • "+N" / "-N"       → apply delta from current  (e.g. "-6" → -6 from current)
//   • clicking ▴ / ▾    → ±1 tick (replicates the native number-input arrows
//                          which we lose by going `type="text"` for delta support)
// On any commit (Enter or blur) we stash the prior value into m._hpPrev
// so the inline ↶ undo button can revert. Undo TOGGLES — clicking it
// swaps current ↔ previous, so repeated clicks bounce between the two.
function renderMonsterHpBar(m, idx) {
  const sn = m.snapshot || {};
  const maxHp = parseInt(sn.hp) || 0;
  const curHp = (typeof m.hp_current === 'number') ? m.hp_current : maxHp;
  const hasPrev = (typeof m._hpPrev === 'number') && m._hpPrev !== curHp;
  const hasBloodiedActions = getWorkshopRenderableActions(sn).some(a => actionBloodiedDamage(a));
  const isBloodied = maxHp > 0 && curHp <= maxHp * 0.5;
  const bloodiedBadge = hasBloodiedActions
    ? `<span class="hpbar-bloodied ${isBloodied ? 'active' : ''}" title="This stat block has HP-threshold damage actions">${isBloodied ? '🩸 Bloodied' : 'Bloodied at ≤50%'}</span>`
    : '';
  return `
    <div class="node-mon-hpbar ${isBloodied && hasBloodiedActions ? 'is-bloodied' : ''}" data-hpbar="${idx}">
      <span class="hpbar-label">HP</span>
      <input type="text" inputmode="numeric"
             class="hpbar-input"
             data-peek-hp="${idx}"
             value="${curHp}"
             title="Type a number to set HP, or ±N (e.g. -6) and Enter to apply a delta">
      <span class="hpbar-ticks">
        <button class="hpbar-tick" data-hp-tick="${idx},1"  type="button" title="+1 HP">▴</button>
        <button class="hpbar-tick" data-hp-tick="${idx},-1" type="button" title="−1 HP">▾</button>
      </span>
      <span class="hpbar-max">/ ${maxHp}</span>
      ${(typeof bloodiedBadge !== "undefined" ? bloodiedBadge : "")}
      <button class="hpbar-undo ${hasPrev ? '' : 'hidden'}"
              data-hp-undo="${idx}" type="button"
              title="${hasPrev ? 'Undo last HP change (back to ' + m._hpPrev + ')' : 'No previous HP value to revert to'}"
              ${hasPrev ? '' : 'disabled'}>↶</button>
    </div>
  `;
}
