// Run in Foundry VTT macro console to export all NPCs at once
const npcs = game.actors.filter(a => a.type === 'npc');
const data = npcs.map(a => a.toJSON());
const blob = new Blob([JSON.stringify(data, null, 2)], {type:'application/json'});
const url = URL.createObjectURL(blob);
const a = document.createElement('a');
a.href = url;
a.download = 'foundry-monsters-export.json';
document.body.appendChild(a);
a.click();
document.body.removeChild(a);
ui.notifications.info(`Exported ${npcs.length} NPCs`);

🔧 Database Schema (one-time setup)

-- Adds three columns to rq_monsters for spell attachment.
-- Safe to run multiple times — uses IF NOT EXISTS.

ALTER TABLE rq_monsters
  ADD COLUMN IF NOT EXISTS spells              jsonb DEFAULT '[]'::jsonb,
  ADD COLUMN IF NOT EXISTS spell_save_dc       int,
  ADD COLUMN IF NOT EXISTS spell_attack_bonus  text DEFAULT '';

-- Optional helpful index for monsters that have spells
CREATE INDEX IF NOT EXISTS rq_monsters_with_spells_idx
  ON rq_monsters ((jsonb_array_length(spells)))
  WHERE jsonb_array_length(spells) > 0;
