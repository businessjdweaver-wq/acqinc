🔮 Import Spells from Foundry

Step 1 — Run the export macro in Foundry

Open your Foundry world's macro directory, create a new Script macro, paste the code below, and execute it. A file called foundry-spells-export.json will download.

📜 Spell-compendium bulk export macro 
// Foundry VTT (D&D 5e) — bulk export of a spell compendium pack.
// Looks for a pack labeled "DDB Spells" by default; falls back to any
// item-pack with "spell" in its label/name. Adjust PREFERRED below if
// your compendium is named differently.
(async () => {
  const PREFERRED = "DDB Spells";
  const allPacks = Array.from(game.packs);
  let pack = allPacks.find(p =>
    (p.metadata?.label || "").toLowerCase() === PREFERRED.toLowerCase()
  );
  if (!pack) {
    pack = allPacks.find(p =>
      p.documentName === "Item" &&
      ((p.metadata?.label || "").toLowerCase().includes("spell") ||
       (p.metadata?.name  || "").toLowerCase().includes("spell"))
    );
  }
  if (!pack) {
    ui.notifications.error("No spell pack found. Available: " +
      allPacks.map(p => p.metadata?.label).join(", "));
    return;
  }
  ui.notifications.info(`Loading "${pack.metadata.label}"…`);
  const docs   = await pack.getDocuments();
  const spells = docs.filter(d => d.type === "spell").map(d => d.toJSON());
  if (!spells.length) { ui.notifications.warn("No spells in pack."); return; }
  const blob = new Blob([JSON.stringify(spells, null, 2)], { type: "application/json" });
  const url  = URL.createObjectURL(blob);
  const a = document.createElement("a");
  a.href = url; a.download = "foundry-spells-export.json";
  document.body.appendChild(a); a.click(); document.body.removeChild(a);
  ui.notifications.info(`Exported ${spells.length} spells from "${pack.metadata.label}"`);
})();

🔧 Database Schema (one-time setup)

-- Run this once in your Supabase SQL editor.
-- Mirrors the rq_monsters pattern: per-user RLS, automatic updated_at.

CREATE TABLE IF NOT EXISTS rq_spells (
  id            uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id       uuid NOT NULL REFERENCES auth.users(id) ON DELETE CASCADE,
  name          text NOT NULL,
  level         int  NOT NULL DEFAULT 0,
  school        text DEFAULT '',
  casting_time  text DEFAULT '',
  range         text DEFAULT '',
  target        text DEFAULT '',
  components    text DEFAULT '',
  duration      text DEFAULT '',
  concentration boolean DEFAULT false,
  ritual        boolean DEFAULT false,
  description   text DEFAULT '',
  higher_levels text DEFAULT '',
  damage        text DEFAULT '',
  damage_type   text DEFAULT '',
  extra         text DEFAULT '',
  extra_type    text DEFAULT '',
  save_ability  text DEFAULT '',
  attack_type   text DEFAULT '',
  scaling       text DEFAULT '',
  source_book   text DEFAULT '',
  source_page   text DEFAULT '',
  classes       text[] DEFAULT '{}',
  tags          text[] DEFAULT '{}',
  foundry_id    text DEFAULT '',
  notes         text DEFAULT '',
  created_at    timestamptz DEFAULT now(),
  updated_at    timestamptz DEFAULT now()
);

ALTER TABLE rq_spells ENABLE ROW LEVEL SECURITY;

CREATE POLICY "rq_spells_owner_all" ON rq_spells FOR ALL
  USING (auth.uid() = user_id) WITH CHECK (auth.uid() = user_id);

CREATE INDEX IF NOT EXISTS rq_spells_user_name_idx
  ON rq_spells (user_id, lower(name));
CREATE INDEX IF NOT EXISTS rq_spells_user_level_idx
  ON rq_spells (user_id, level);

CREATE OR REPLACE FUNCTION rq_spells_touch_updated_at()
  RETURNS trigger AS $$
  BEGIN NEW.updated_at = now(); RETURN NEW; END;
$$ LANGUAGE plpgsql;

DROP TRIGGER IF EXISTS rq_spells_touch_updated_at ON rq_spells;
CREATE TRIGGER rq_spells_touch_updated_at
  BEFORE UPDATE ON rq_spells
  FOR EACH ROW EXECUTE FUNCTION rq_spells_touch_updated_at();
